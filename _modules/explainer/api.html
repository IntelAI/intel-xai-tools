import sys
from subprocess import run, CalledProcessError, CompletedProcess, SubprocessError
import shutil
from abc import ABC
from typing import Any, Callable

from . import ExplainerLoader, ExplainerModuleSpec, EXT_YAML

explainers_folder = os.path.abspath(
    os.path.join(os.path.dirname(__file__), "explainers"))



[docs]class Explainer(ABC):
    """Explainer API base class
    """

    def __init__(self,  model: Any = None) -> None:
        """takes any type of model or None

        Args:
            model (Any): any instance of any type of model
        """
        self._explainer: Callable
        self._model: Any = model

    def __call__(self, **kwargs):
        """allows arguments to be curried into a context

        Returns:
            _type_: _description_
        """
        if "model" in kwargs:
            self._model = kwargs["model"]

        return self

    @property
    def explainables(self) -> list[str]:
        """Return the list of explainable yaml files

        Returns:
            list: list of explainables
        """
        suffix = ".yaml"
        r_var: list = []
        for filename in os.listdir(explainers_folder):
            if filename.endswith(suffix):
                r_var.append(filename[0:-len(suffix)])
        r_var.sort()
        return r_var

    @property
    def explainer(self) -> Any:
        """Loads the explainer and returns it

        Returns:
            Any: any explainer for now
        """
        return self._explainer


[docs]    def export_to(self, yamlpath: str) -> ExplainerModuleSpec:
        """creates a pkz file under explainer/explainers with an optional entry_point

        Args:
            yamlpath (str): the yaml file path

        Returns:
            ExplainerModuleSpec: A ModuleSpec subclass
        """
        def check_for_errors(cmd, sub_cmd, error):
            if error is not None:
                print(f"{error}")
            if sub_cmd is not None and sub_cmd.stderr is not None:
                print(f"{cmd}: {sub_cmd.stderr}")

        fullpath = os.path.abspath(os.path.join(explainers_folder, yamlpath+EXT_YAML))
        module: ExplainerModuleSpec = None
        if os.path.exists(fullpath):
            exl = ExplainerLoader(fullpath)
            module = exl.create_module()
            spec = module.spec
            if hasattr(spec, "plugin"):
                if hasattr(spec, "dependencies"):
                    parts: list[str] = os.path.splitext(spec.plugin)
                    dirname: str = parts[0]
                    dirpath: str = os.path.join(os.path.dirname(fullpath), dirname)
                    if os.path.exists(dirpath) is False:
                        os.mkdir(dirpath)
                        if os.path.exists(dirpath):
                            install_cmd: CompletedProcess = None
                            cmd = f"{sys.executable} -m pip install "
                            cmd += " ".join(str(x) for x in spec.dependencies)
                            cmd += f" --target {dirpath}"
                            try:
                                install_cmd = run(
                                    cmd.split(), capture_output=True,
                                    text=True)
                                install_cmd.check_returncode()
                                if hasattr(spec, "entry_point"):
                                    entry_point = os.path.join(dirpath, "entry_point.py")
                                    with open(entry_point, 'w', encoding='utf-8') as ep_file:
                                        ep_file.write(spec.entry_point)
                                shutil.make_archive(
                                    dirname, format="zip", root_dir=dirpath)
                                zipfile = dirname+".zip"
                                currentpath: str = os.path.join(os.curdir, zipfile)
                                targetpath: str = os.path.join(os.path.dirname(fullpath), spec.plugin)
                                shutil.move(currentpath, targetpath)
                            except CalledProcessError as cpe:
                                check_for_errors(cmd, install_cmd, cpe)
                            except SubprocessError as sube:
                                check_for_errors(cmd, install_cmd, sube)
                            except ValueError as vae:
                                check_for_errors(cmd, install_cmd, vae)
                            finally:
                                shutil.rmtree(dirpath, ignore_errors=True)
        else:
            print(f"{fullpath}: does not exit", file=sys.stderr)

        return module



[docs]    def import_from(self, yamlpath: str) -> ExplainerModuleSpec:
        """Import a yaml file using the ExplainerLoader

        Args:
            yamlpath (str): path to the yaml file

        Returns:
            ExplainerModuleSpec: A ModuleSpec subclass
        """
        fullpath = os.path.abspath(os.path.join(explainers_folder, yamlpath+EXT_YAML))
        module: ExplainerModuleSpec = None
        if os.path.exists(fullpath):
            exl = ExplainerLoader(fullpath)
            module = exl.create_module()
        else:
            print(f"{fullpath}: does not exit", file=sys.stderr)
        return module
